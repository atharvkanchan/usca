    num = df.select_dtypes(include=['number']).columns.tolist()
    target = st.selectbox("Target", num)

===== _4_Forecasting.py =====
import streamlit as st
import pandas as pd
import matplotlib.pyplot as plt

def show():
    st.title("ðŸ“… Forecasting")
    if "df" not in st.session_state:
        try: st.session_state["df"] = pd.read_csv("india_supply_chain_2024_2025.csv")
        except: st.error("Dataset missing"); return
    df = st.session_state["df"]
    date_col = None
    for c in df.columns:
        if "date" in c.lower(): date_col = c; break
    df[date_col] = pd.to_datetime(df[date_col], errors="coerce")
    num = df.select_dtypes(include=['number']).columns.tolist()
    val = st.selectbox("Value", num)
    ts = df.set_index(date_col)[val].asfreq("D").interpolate()
    fig, ax = plt.subplots()
    ax.plot(ts.index, ts.values)
    st.pyplot(fig)
